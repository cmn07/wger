28class Command(BaseCommand): 

29 help = 'Resets the cache for all routines.' 

30 

31 def handle(self, *args, **options): 

32 total_routines = Routine.objects.count() 

33 self.stdout.write(f'Updating cache for {total_routines} routines...') 

34 

35 with tqdm(total=total_routines, unit='routine', unit_scale=True) as pbar: 

36 routine: Routine 

37 for routine in Routine.objects.all(): 

38 reset_routine_cache(routine) 

39 _ = routine.date_sequence 

40 pbar.update(1) 

41 

42 self.stdout.write(self.style.SUCCESS('Successfully updated cache for all routines!'))
