38def fetch_github_stats() -> dict: 

39 context = cache.get(CACHE_KEY) 

40 if context: 

41 return context 

42 

43 context = { 

44 'nr_users': 1, 

45 'nr_exercises': 1, 

46 'nr_ingredients': 1, 

47 'nr_stars': 1, 

48 } 

49 

50 try: 

51 result_github_api = requests.get('https://api.github.com/repos/wger-project/wger').json() 

52 context['nr_users'] = User.objects.count() 

53 context['nr_exercises'] = Exercise.objects.count() 

54 context['nr_ingredients'] = Ingredient.objects.count() 

55 context['nr_stars'] = result_github_api.get('stargazers_count', '2000') 

56 cache.set(CACHE_KEY, context, 60 * 60 * 24 * 7) # one week 

57 except Exception as e: 

58 logger.error(f'Error fetching github stats: {e}') 

59 

60 return context

63def features(request): 

64 """ 

65 Render the landing page 

66 """ 

67 

68 context = fetch_github_stats() 

69 

70 form = WgerSignupForm() 

71 form.fields['username'].widget.attrs.pop('autofocus', None) 

72 

73 context['form'] = form 

74 context['allow_registration'] = settings.WGER_SETTINGS['ALLOW_REGISTRATION'] 

75 context['allow_guest_users'] = settings.WGER_SETTINGS['ALLOW_GUEST_USERS'] 

76 

77 return render(request, 'features.html', context)
