31class Command(WgerCommand): 

32 """ 

33 Synchronizes exercise data from a wger instance to the local database 

34 """ 

35 

36 remote_url = settings.WGER_SETTINGS['WGER_INSTANCE'] 

37 

38 help = """Synchronizes exercise data from a wger instance to the local database. 

39 This script also deletes entries that were removed on the server such 

40 as exercises, images or videos. 

41 

42 Please note that at the moment the following objects can only identified 

43 by their id. If you added new objects they might have the same IDs as the 

44 remote ones and will be overwritten: 

45 - categories 

46 - muscles 

47 - equipment 

48 """ 

49 

50 def add_arguments(self, parser): 

51 super().add_arguments(parser) 

52 

53 parser.add_argument( 

54 '--dont-delete', 

55 action='store_true', 

56 dest='skip_delete', 

57 default=False, 

58 help='Skips deleting any entries', 

59 ) 

60 

61 def handle(self, **options): 

62 super().handle(**options) 

63 

64 # Process everything 

65 sync_languages(self.stdout.write, self.remote_url, self.style.SUCCESS) 

66 sync_categories(self.stdout.write, self.remote_url, self.style.SUCCESS) 

67 sync_muscles(self.stdout.write, self.remote_url, self.style.SUCCESS) 

68 sync_equipment(self.stdout.write, self.remote_url, self.style.SUCCESS) 

69 sync_licenses(self.stdout.write, self.remote_url, self.style.SUCCESS) 

70 sync_exercises(self.stdout.write, self.remote_url, self.style.SUCCESS) 

71 if not options['skip_delete']: 

72 handle_deleted_entries(self.stdout.write, self.remote_url, self.style.SUCCESS)
