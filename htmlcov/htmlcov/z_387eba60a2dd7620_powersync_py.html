16""" 

17Shared infrastructure for the PowerSync push handlers. 

18 

19Each app declares one ``PowerSyncHandler`` subclass per synchronised table 

20and registers it via ``@register_handler``. The dispatcher in 

21``wger.core.api.views.upload_powersync_data`` looks up the handler by table 

22name and calls ``dispatch(http_verb, ...)``. 

23""" 

24

26import logging 

27from typing import Any 

28 

29# Django 

30from django.db.models import Model 

31 

32# Third Party 

33from rest_framework.serializers import Serializer

35# wger 

36from wger.utils.viewsets import check_fk_ownership

39logger = logging.getLogger(__name__)

41 

42REGISTRY: dict[str, 'PowerSyncHandler'] = {}

45class PowerSyncHandler: 

46 """ 

47 Base class for PowerSync upload handlers. 

48 

49 Subclasses set declarative attributes describing the table, model, 

50 serializer and ownership rules. The most common variations 

51 (``lookup_field``, ``user_filter``, FK-ownership viewset, serializer 

52 context, save kwargs, payload preprocessing) are all expressible via 

53 attributes or single-method overrides. 

54 """ 

55 

56 # Django model and DRF serializer for the table. 

57 model: type[Model] = None 

58 serializer_class: type[Serializer] = None 

59 

60 # PowerSync table name. Populated by ``register_handler`` from 

61 # ``model._meta.db_table`` — the same value PowerSync sees in Postgres. 

62 table: str = '' 

63 

64 # Set to a ViewSet class to enable check_fk_ownership against 

65 # ``ViewSetClass.get_owner_objects()``. Leave as None if the model has 

66 # no FKs that need ownership-validation (e.g. WeightEntry, Category). 

67 viewset_class = None 

68 

69 # Lookup config for update/delete: 

70 # Model.objects.get(<lookup_field>=payload['id'], <user_filter>=user_id) 

71 # ``lookup_field`` is 'pk' (most models) or 'uuid' (WeightEntry). 

72 # ``user_filter`` is 'user_id' for direct ownership or a chained lookup 

73 # like 'plan__user_id' / 'meal__plan__user_id' / 'category__user_id'. 

74 lookup_field: str = 'pk' 

75 user_filter: str = 'user_id' 

76 

77 # Set to False for tables where creation must go through REST (Routine, 

78 # Image). The dispatcher then returns "Method not allowed" instead of 

79 # running the create handler. 

80 supports_create: bool = True 

81 

82 # Set to False for tables that must never be deleted through PowerSync 

83 # (UserProfile). handle_delete then returns "Method not allowed" instead 

84 # of deleting the row. 

85 supports_delete: bool = True 

86 

87 # If True, the handler passes ``user_id`` into the serializer context so 

88 # that owner-scoped fields (e.g. WorkoutLogSerializer.session) can filter 

89 # their queryset. 

90 pass_user_id_in_context: bool = False 

91 

92 @property 

93 def label(self) -> str: 

94 """Human-readable model name used in log lines and error details.""" 

95 return self.model.__name__ 

96

98 # Hooks 

99 # 

100 

101 def preprocess_payload(self, payload: dict[str, Any]) -> dict[str, Any]: 

102 """ 

103 Last-chance mutation of the payload before validation. Default is a 

104 no-op; gallery uses this to drop the binary ``image`` field that only 

105 REST is allowed to write. 

106 """ 

107 return payload 

108 

109 def create_save_kwargs(self, payload: dict[str, Any], user_id: int) -> dict[str, Any]: 

110 """ 

111 kwargs forwarded to ``serializer.save()`` on create. Default forces the 

112 owning user. Override for client-supplied UUID PKs (WeightEntry), 

113 ordering (Meal/MealItem) or models that don't take ``user_id`` because 

114 ownership rides on a FK chain (LogItem, MealItem). 

115 """ 

116 return {'user_id': user_id} 

117 

118 # 

119 # Public CRUD entry points 

120 # 

121 

122 def handle_create(self, payload: dict[str, Any], user_id: int) -> dict | None: 

123 if not self.supports_create: 

124 return { 

125 'error': 'Method not allowed', 

126 'details': f'{self.label} creation must go through the REST API', 

127 } 

128 

129 logger.debug(f'Received PowerSync payload for {self.label} create: {payload}') 

130 payload = self.preprocess_payload(payload) 

131 

132 # Idempotent create: a redelivered PUT for a row this user already owns is 

133 # a no-op success, not a PK-conflict error (PowerSync redelivers writes). 

134 if 'id' in payload and self._get_or_none(payload, user_id) is not None: 

135 return self._ack_existing(payload['id']) 

136 

137 fk_err = self._check_fk(payload, user_id) 

138 if fk_err is not None: 

139 return fk_err 

140 

141 serializer = self.serializer_class( 

142 data=payload, 

143 context=self._serializer_context(user_id), 

144 ) 

145 if serializer.is_valid(): 

146 serializer.save(**self.create_save_kwargs(payload, user_id)) 

147 logger.debug(f'Created {self.label} {payload.get("id")} for user {user_id}') 

148 return None 

149 return self._validation_failed(serializer.errors) 

150 

151 def handle_update(self, payload: dict[str, Any], user_id: int) -> dict | None: 

152 logger.debug(f'Received PowerSync payload for {self.label} update: {payload}') 

153 payload = self.preprocess_payload(payload) 

154 

155 entry = self._get_or_none(payload, user_id) 

156 if entry is None: 

157 return self._ack_missing('update', payload['id']) 

158 

159 fk_err = self._check_fk(payload, user_id) 

160 if fk_err is not None: 

161 return fk_err 

162 

163 serializer = self.serializer_class( 

164 entry, 

165 data=payload, 

166 partial=True, 

167 context=self._serializer_context(user_id), 

168 ) 

169 if serializer.is_valid(): 

170 serializer.save() 

171 logger.debug(f'Updated {self.label} {entry.pk} for user {user_id}') 

172 return None 

173 return self._validation_failed(serializer.errors) 

174 

175 def handle_delete(self, payload: dict[str, Any], user_id: int) -> dict | None: 

176 if not self.supports_delete: 

177 return { 

178 'error': 'Method not allowed', 

179 'details': f'{self.label} cannot be deleted', 

180 } 

181 

182 logger.debug(f'Received PowerSync payload for {self.label} delete: {payload}') 

183 entry = self._get_or_none(payload, user_id) 

184 if entry is None: 

185 return self._ack_missing('delete', payload['id']) 

186 entry.delete() 

187 logger.debug(f'Deleted {self.label} {payload["id"]} for user {user_id}') 

188 return None 

189 

190 def dispatch( 

191 self, 

192 http_verb: str, 

193 payload: dict[str, Any], 

194 user_id: int, 

195 ) -> dict | None: 

196 """Route an HTTP verb to the matching CRUD handler.""" 

197 if http_verb == 'PUT': 

198 return self.handle_create(payload, user_id) 

199 if http_verb == 'PATCH': 

200 return self.handle_update(payload, user_id) 

201 if http_verb == 'DELETE': 

202 return self.handle_delete(payload, user_id) 

203 return {'error': 'Method not allowed', 'details': f'Unsupported HTTP verb: {http_verb}'} 

204

206 # Internals 

207 # 

208 

209 def _serializer_context(self, user_id: int) -> dict[str, Any]: 

210 if self.pass_user_id_in_context: 

211 return {'user_id': user_id} 

212 return {} 

213 

214 def _check_fk(self, payload: dict[str, Any], user_id: int) -> dict | None: 

215 if self.viewset_class is None: 

216 return None 

217 if check_fk_ownership(payload, self.viewset_class.get_owner_objects(), user_id): 

218 return None 

219 return { 

220 'error': 'Forbidden', 

221 'details': f'{self.label} references an object you do not own', 

222 } 

223 

224 def _get_or_none(self, payload: dict[str, Any], user_id: int): 

225 ident = payload['id'] 

226 try: 

227 return self.model.objects.get(**{self.lookup_field: ident, self.user_filter: user_id}) 

228 except self.model.DoesNotExist: 

229 return None 

230 

231 def _ack_missing(self, verb: str, ident) -> None: 

232 """ 

233 Acknowledge a write whose target row is gone as an idempotent no-op. 

234 

235 PowerSync redelivers operations and is server-authoritative, so a missing 

236 target (already deleted, or a redelivered op) is the desired end state, 

237 not an error: returning an error would surface a spurious failure on the 

238 client and block nothing useful. This is also what a cross-user write 

239 resolves to (the owner filter matches nothing), which stays a safe no-op. 

240 """ 

241 logger.info(f'PowerSync {verb} for missing {self.label} {ident}, acknowledged as no-op') 

242 return None 

243 

244 def _ack_existing(self, ident) -> None: 

245 """ 

246 Acknowledge a redelivered create as an idempotent no-op. 

247 

248 The row already exists and is owned by this user, so re-inserting it would 

249 only raise a PK/unique conflict. PowerSync may redeliver the same create, 

250 so this is the desired end state, not an error. 

251 """ 

252 logger.info(f'PowerSync create for existing {self.label} {ident}, acknowledged as no-op') 

253 return None 

254 

255 def _validation_failed(self, errors) -> dict: 

256 logger.warning(f'PowerSync {self.label} validation failed: {errors}') 

257 return {'error': 'Validation failed', 'details': errors}

260def register_handler(cls: type[PowerSyncHandler]) -> type[PowerSyncHandler]: 

261 """ 

262 Class decorator: instantiate ``cls`` and add the singleton to ``REGISTRY`` 

263 keyed by the model's DB table name (which is what the PowerSync sync rules 

264 expose). 

265 """ 

266 if cls.model is None: 

267 raise ValueError(f'{cls.__name__} must set `model`') 

268 cls.table = cls.model._meta.db_table 

269 if cls.table in REGISTRY: 

270 raise ValueError( 

271 f'PowerSync handler for table "{cls.table}" is already registered ' 

272 f'(by {type(REGISTRY[cls.table]).__name__})' 

273 ) 

274 REGISTRY[cls.table] = cls() 

275 return cls
